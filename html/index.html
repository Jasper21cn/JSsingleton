ECMAScript 6 Singleton Example 
                
    let instance = null; // Default value

    class Singleton {
        constructor () {
            if(!instance){
                instance = this;
            }
            this.createInstance = new Object('my Value'); // give a property for testing
            return instance;
        }
    }

    let instance1 = new Singleton();
    let instance2 = new Singleton();
    console.log(instance1.createInstance === instance2.createInstance);  // return true
